28pub fn read_file_content(path: &Path) -> Result<String> {
29    fs::read_to_string(path)
30        .map_err(|e| QRCryptError::Io(e))
31        .map(|content| content.trim().to_string())
32}

34pub fn write_file_content(path: &Path, content: &str) -> Result<()> {
35    if let Some(parent) = path.parent() {
36        fs::create_dir_all(parent).map_err(QRCryptError::Io)?;
37    }
38    
39    fs::write(path, content).map_err(QRCryptError::Io)
40}

42pub fn load_shares_from_files(file_paths: &[impl AsRef<Path>]) -> Result<Vec<ShamirShare>> {
43    let mut shares = Vec::new();
44    
45    for path in file_paths {
46        let content = read_file_content(path.as_ref())?;
47        
48        // Try to parse as QRData first (from QR code scan), then as direct ShamirShare
49        let share = if let Ok(qr_data) = QRReader::parse_qr_data(&content) {
50            QRReader::parse_shamir_share(&qr_data)?
51        } else {
52            serde_json::from_str::<ShamirShare>(&content)
53                .map_err(|e| QRCryptError::InvalidInput(format!("Invalid share file format: {}", e)))?
54        };
55        
56        shares.push(share);
57    }
58    
59    Ok(shares)
60}

102pub fn validate_seed_phrase(phrase: &str) -> Result<()> {
103    let words: Vec<&str> = phrase.split_whitespace().collect();
104    
105    if words.is_empty() {
106        return Err(QRCryptError::InvalidInput("Seed phrase cannot be empty".to_string()));
107    }
108    
109    // Basic validation for common seed phrase lengths
110    let word_count = words.len();
111    if ![12, 15, 18, 21, 24].contains(&word_count) {
112        println!("Warning: Seed phrase has {} words. Common lengths are 12, 15, 18, 21, or 24 words.", word_count);
113    }
114    
115    // Check for very short words that might indicate input errors
116    for word in &words {
117        if word.len() < 2 {
118            return Err(QRCryptError::InvalidInput(format!("Word '{}' seems too short. Please check your input.", word)));
119        }
120    }
121    
122    Ok(())
123}

125pub fn generate_example_seed(word_count: u8) -> Result<String> {
126    // Generate a deterministic but realistic-looking seed phrase for testing
127    let words = [
128        "abandon", "ability", "able", "about", "above", "absent", "absorb", "abstract",
129        "absurd", "abuse", "access", "accident", "account", "accuse", "achieve", "acid",
130        "acoustic", "acquire", "across", "act", "action", "actor", "actress", "actual",
131        "adapt", "add", "addict", "address", "adjust", "admit", "adult", "advance",
132        "advice", "aerobic", "affair", "afford", "afraid", "again", "agent", "agree",
133        "ahead", "aim", "air", "airport", "aisle", "alarm", "album", "alcohol"
134    ];
135    
136    if ![12, 15, 18, 21, 24].contains(&word_count) {
137        return Err(QRCryptError::InvalidInput("Word count must be 12, 15, 18, 21, or 24".to_string()));
138    }
139    
140    let selected_words: Vec<&str> = words.iter()
141        .take(word_count as usize - 1)
142        .copied()
143        .collect();
144    
145    let mut result = selected_words.join(" ");
146    result.push_str(" about"); // Always end with "about" for consistency
147    
148    Ok(result)
149}
