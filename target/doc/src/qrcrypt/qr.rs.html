77    fn create_qr_image(&self, data: &str) -> Result<DynamicImage> {
78        let code = QrCode::with_error_correction_level(data, self.error_correction)
79            .map_err(|e| QRCryptError::QRGeneration(format!("Failed to create QR code: {:?}", e)))?;
80
81        // Use char as the pixel type which is supported by qrcode
82        let string_image = code.render::<char>()
83            .light_color(' ')
84            .dark_color('█')
85            .build();
86
87        // Convert the string representation to a pixel-based image
88        let lines: Vec<&str> = string_image.lines().collect();
89        let height = lines.len() as u32;
90        let width = if height > 0 { lines[0].len() as u32 } else { 0 };
91
92        if width == 0 || height == 0 {
93            return Err(QRCryptError::QRGeneration("Generated QR code has zero dimensions".to_string()));
94        }
95
96        let mut pixels = Vec::new();
97        for line in &lines {
98            for ch in line.chars() {
99                pixels.push(if ch == ' ' { 255u8 } else { 0u8 }); // White for space, black for █
100            }
101        }
102
103        // Scale the image
104        let scaled_width = width * self.scale;
105        let scaled_height = height * self.scale;
106        let mut scaled_pixels = Vec::with_capacity((scaled_width * scaled_height) as usize);
107
108        for y in 0..scaled_height {
109            for x in 0..scaled_width {
110                let original_x = x / self.scale;
111                let original_y = y / self.scale;
112                let original_index = (original_y * width + original_x) as usize;
113                if original_index < pixels.len() {
114                    scaled_pixels.push(pixels[original_index]);
115                } else {
116                    scaled_pixels.push(255u8);
117                }
118            }
119        }
120
121        let img_buffer = image::ImageBuffer::from_raw(scaled_width, scaled_height, scaled_pixels)
122            .ok_or_else(|| QRCryptError::QRGeneration("Failed to create image buffer".to_string()))?;
123
124        Ok(DynamicImage::ImageLuma8(img_buffer))
125    }

138    pub fn save_shamir_qrs(
139        &self,
140        shares: &[ShamirShare],
141        base_path: &Path,
142        prefix: &str,
143    ) -> Result<Vec<String>> {
144        let mut file_paths = Vec::new();
145        
146        for (i, share) in shares.iter().enumerate() {
147            let filename = format!("{}_{}_of_{}_share_{}.png", 
148                prefix, 
149                i + 1, 
150                shares.len(),
151                share.share_id
152            );
153            let file_path = base_path.join(&filename);
154            
155            let image = self.generate_shamir_qr(share)?;
156            self.save_qr_image(&image, &file_path)?;
157            
158            file_paths.push(filename);
159        }
160        
161        Ok(file_paths)
162    }

189pub struct QRReader;
190
191impl QRReader {
192    pub fn parse_qr_data(json_data: &str) -> Result<QRData> {
193        serde_json::from_str(json_data)
194            .map_err(|e| QRCryptError::QRParsing(format!("Failed to parse QR data: {}", e)))
195    }
196
197    pub fn parse_encrypted_data(qr_data: &QRData) -> Result<EncryptedData> {
198        match qr_data.data_type {
199            QRDataType::EncryptedSecret => {
200                serde_json::from_str(&qr_data.content)
201                    .map_err(|e| QRCryptError::QRParsing(format!("Failed to parse encrypted data: {}", e)))
202            }
203            _ => Err(QRCryptError::QRParsing("QR code does not contain encrypted data".to_string())),
204        }
205    }
206
207    pub fn parse_shamir_share(qr_data: &QRData) -> Result<ShamirShare> {
208        match qr_data.data_type {
209            QRDataType::ShamirShare => {
210                serde_json::from_str(&qr_data.content)
211                    .map_err(|e| QRCryptError::QRParsing(format!("Failed to parse Shamir share: {}", e)))
212            }
213            _ => Err(QRCryptError::QRParsing("QR code does not contain Shamir share data".to_string())),
214        }
215    }
